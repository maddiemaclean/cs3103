2. Create a stored procedure to add quotes to the database

DELIMITER //
DROP PROCEDURE IF EXISTS "addQuote" //

CREATE PROCEDURE addQuote ( textIn VARCHAR (250))
begin
    INSERT INTO QUOTES (textIn)
    VALUES (textIn);
end//
DELIMITER ;

3. Create a stored procedure to randonly access a quote from the table

DELIMITER //
DROP PROCEDURE IF EXISTS "getQuote" //

CREATE PROCEDURE getQuote ()
begin
    SELECT* FROM quotes ORDER by RAND() LIMIT 1;
end//
DELIMITER ;
